堆排序

fixup()函数：向上调整节点 

    //@function：  向上调整节点
    //@input  a：  堆的头结点
    //@input  len：堆的长度
    void fixup(int *a, int len)
    {
        int curr = len -1;
        int parent = (curr-1)/2;
        while(parent>=0 && a[parent] > a[curr])
        {
            swap(a[curr], a[parent]);
            curr = parent;
            parent = (curr-1)/2;
        }
    }


fixdown()函数：向下调整节点 

    //@function：  向下调整节点
    //@input  a：  堆的头结点
    //@input  i：  待调整结点的下标
    //@input  len：堆的长度
    void fixdown(int *a, int i, int len)
    {
        if(len == 1)
            return;

        int last = len-1;
        while(i <= (last-1)/2)
        {
            int son;

            if(2*i+2 < last)
                son = a[2*i+1] < a[2*i+2] ? 2*i+1 : 2*i+2;
            else
                son = 2*i+1;

            if(a[i] > a[son])
            {
                swap(a[i], a[son]);
                i = son;
            }
            else
                break;
        }
    }


del()函数：删除节点 

    //@function：  删除节点
    //@input  a：  堆的头结点
    //@input  len：堆的长度
   void del(int *a, int &len)
    {
        swap(a[0], a[len-1]);
        len--;
        fixdown(a, 0, len);
    }


add()函数：插入节点 

    //@function：  插入节点
    //@input  a：  堆的头结点
    //@input  len：堆的长度
    void add(int *a, int &len, int num)
    {
        a[len] = num;
        len++;
        fixup(a, len);
    }


build()函数：建树 

    //@function：  建树
    //@input  a：  堆的头结点
    //@input  len：堆的长度
   void build(int *a, int &len)
    {
        for(int i = (len-1)/2; i>=0; i--)
            fixdown(a, i, len);
    }
