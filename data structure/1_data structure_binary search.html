带头结点的单链表

  • 实现方法：
    　　插入节点
     void insert(int num, node * head)
  {
      while(head->next != NULL)
          head = head->next;
      head->next = new node(num);
  }

    　　删除节点
     void del(int num, node * head)
  {
      while(head->next != NULL)
      {
          node * current = head->next;
          if(head->next->val == num)
          {
              head->next = head->next->next;
              delete current;
          }
          head = head->next; 
      }
  }

    　　遍历链表
      void print(node * head)
  {
      while(head != NULL)
      {
          cout<<head->val<<endl;
          head = head->next;
      }
  }
