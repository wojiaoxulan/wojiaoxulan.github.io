编程珠玑第一章

  • 位向量： 由二进制位组成的向量，直接对每一个bit进行操作。
    　　　 C++语言使用基本类型int或者char等来存储数据，但是如果想要记录每个对象的状态0或者1，那么使用基本类型int或者char 会浪费空间。而如果使用位向量，那么可以充分利用内存。
    　　　 首先定义数据类型：

      #define N           10000000  //number of elements
  #define BITPERWORD  32        //bits of int depends on machine
  unsigned int a[100];

    　　　直接使用整型数组作为基本的数据类型，而数组的大小和要存储的元素数量相关。 位向量应该支持三种操作：对某一位置1、置0、读取。
    　　　 将第num个bit位置1

      void betrue(int num)
  {   a[num>>5] |= 1 << (num & 31);          }

    　　　 将第num个bit位置0

      void befalse(int num)
  {   a[num>>5] &= ~(1 << (num & 31));       }

    　　　 对第num个bit读值

      unsigned int read(int num)
  {   return a[num>>5] & (1 << (num & 31));  }


    　　　 下面是使用位向量实现排序的例子。

      #include "stdafx.h"
  #include<string>
  #include<iostream>
  using namespace std;
  unsigned int a[100];
  void betrue(int num);         //设第num个bit位是1
  void befalse(int num);        //设第num个bit位是0
  unsigned int read(int num);  //返回第num个bit位的值
  int main()
  {

      for(int i=0; i<100; i++)
          a[i] = 0;
      int n;
      cin>>n;
      int num;
      while(n--)
      {
          cin>>num;
          betrue(num);
      }
      for(int k=0;k<3200;k++)
             if(read(k))
              cout<<k<<endl;
   return 0;
  }

  void betrue(int num)
  {   a[num>>5] |= 1 << (num & 31);          }

  void befalse(int num)
  {   a[num>>5] &= ~(1 << (num & 31));       }

  unsigned int read(int num)
  {   return a[num>>5] & (1 << (num & 31));  }
