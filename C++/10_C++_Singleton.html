单例模式

定义：
　　单例模式就是一个类只能有一个实例，这种类的实例要求是唯一的， 

#include "stdafx.h"
#include "iostream"
#include "cstdlib"
using namespace std;

class Singleton    
{  
    public:  
        static Singleton& GetInstance(); 
        int val; 
    private:
        static int num;
        Singleton()    { num++; val=12345;} 
        Singleton(const Singleton &copy) { cout<<"客官不可以"<<endl;} 
        static Singleton *singleton;  
};  

int Singleton::num = 0;
Singleton* Singleton::singleton = NULL;  
Singleton& Singleton::GetInstance()  
{  
    if(singleton == NULL)  
        singleton = new Singleton();  
    else
        cout<<"faiLure"<<endl;
    return *singleton;  
}  

int main(int argc, char* argv[])
{
    Singleton &a = Singleton::GetInstance();
    cout<<a.val<<endl;
    return 0;
}
